* Tangle files

Babel tangle files from the terminal.

#+begin_src shell

  for i in *.org; do
      [ -f "$i" ] || break
      emacs --batch --eval "(require 'org)" --eval "(org-babel-tangle-file \"$i\")"
  done

#+end_src

#+begin_src shell

#+end_src
